PLUS MINUS 

#include <stdio.h>
void plusMinus(int numbers[], int size);
int main() {
    int size;
    scanf("%d", &size);
    if (size <= 0) {
        printf("Array size must be greater than 0.\n");
        return 1;
    }
    int numbers[size];
    for (int i = 0; i < size; i++) {
        scanf("%d", &numbers[i]);
    }
    plusMinus(numbers, size);
    return 0;
}
void plusMinus(int numbers[], int size) {
    int countPos = 0, countNeg = 0, countZero = 0;
    for (int i = 0; i < size; i++) {
        if (numbers[i] > 0)
            countPos++;
        else if (numbers[i] < 0)
            countNeg++;
        else
            countZero++;
    }
    printf("%.6f\n", (double)countPos / size);
    printf("%.6f\n", (double)countNeg / size);
    printf("%.6f\n", (double)countZero / size);
}
